    <script>
        let selectedTheme = null;

        function selectTheme(element, themeId) {
            // Remove selected state from all themes
            document.querySelectorAll('.theme-card').forEach(card => {
                card.classList.remove('selected');
                const indicator = card.querySelector('.w-5, .w-6');
                if (indicator) {
                    indicator.classList.remove('border-primary-500', 'bg-primary-500');
                    indicator.classList.add('border-gray-500');
                    indicator.innerHTML = '';
                }
            });

            // Add selected state to clicked theme
            element.classList.add('selected');
            const indicator = element.querySelector('.w-5, .w-6');
            if (indicator) {
                indicator.classList.remove('border-gray-500');
                indicator.classList.add('border-primary-500', 'bg-primary-500');
                indicator.innerHTML = '<svg class="w-3 h-3 text-white" fill="currentColor" viewBox="0 0 20 20"><path fill-rule="evenodd" d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z" clip-rule="evenodd"></path></svg>';
            }

            selectedTheme = themeId;
        }

        function toggleSwitch(element) {
            element.classList.toggle('active');
        }

        function toggleCommunityOptions() {
            const options = document.getElementById('community-options');
            if (options.classList.contains('hidden')) {
                options.classList.remove('hidden');
                setTimeout(() => {
                    options.style.opacity = '1';
                    options.style.transform = 'translateY(0)';
                }, 10);
            } else {
                options.style.opacity = '0';
                options.style.transform = 'translateY(-10px)';
                setTimeout(() => {
                    options.classList.add('hidden');
                }, 300);
            }
        }

        function updateCharCount(textarea, counterId) {
            const counter = document.getElementById(counterId);
            const current = textarea.value.length;
            const max = textarea.maxLength;
            counter.textContent = `${current}/${max}`;
            
            if (current > max * 0.9) {
                counter.classList.add('text-red-400');
            } else {
                counter.classList.remove('text-red-400');
            }
        }

        function toggleUploadMode(mode) {
            const uploadArea = document.getElementById('upload-area');
            if (mode === 'url') {
                uploadArea.innerHTML = `
                    <div class="space-y-6">
                        <div class="w-16 h-16 bg-gradient-to-br from-primary-500 to-primary-600 rounded-2xl flex items-center justify-center mx-auto">
                            <svg class="w-8 h-8 text-white" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M13.828 10.172a4 4 0 00-5.656 0l-4 4a4 4 0 105.656 5.656l1.102-1.101m-.758-4.899a4 4 0 005.656 0l4-4a4 4 0 00-5.656-5.656l-1.1 1.1"></path>

                            Add URL

            } else {
                uploadArea.innerHTML = `
                    <div class="mb-6">
                        <div class="w-16 h-16 bg-gradient-to-br from-primary-500 to-primary-600 rounded-2xl flex items-center justify-center mx-auto mb-4">

                    <input type="file" class="hidden" accept="image/*,.gif" onchange="handleFileUpload(this)">

            }

        function handleFileUpload(input) {
            const file = input.files[0];
            if (file) {
                const uploadArea = input.closest('.upload-zone');
                uploadArea.innerHTML = `
                    <div class="flex items-center justify-between p-4 glass rounded-xl">
                        <div class="flex items-center space-x-4">
                            <div class="w-12 h-12 bg-gradient-to-br from-green-500 to-emerald-500 rounded-xl flex items-center justify-center">

                                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 13l4 4L19 7"></path>
                                </svg>
                            </div>
                            <div>
                                <p class="font-semibold">${file.name}</p>
                                <p class="text-sm text-gray-400">${(file.size / 1024 / 1024).toFixed(2)} MB</p>
                            </div>
                        </div>
                        <button type="button" class="text-gray-400 hover:text-red-400 transition-colors" onclick="clearUpload(this)">
                            <svg class="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24">

                            </svg>

                    </div>
                `;
            }
        }

            uploadArea.innerHTML = `

        // Enhanced drag and drop
        document.addEventListener('click', function(e) {
            if (e.target.closest('.upload-zone')) {
                const fileInput = e.target.closest('.upload-zone').querySelector('input[type="file"]');
                if (fileInput) {
                    fileInput.click();
                }
            }
        });

        document.addEventListener('dragover', function(e) {
            e.preventDefault();
            const uploadArea = e.target.closest('.upload-zone');
            if (uploadArea) {
                uploadArea.classList.add('dragover');
            }
        });

        document.addEventListener('dragleave', function(e) {
            const uploadArea = e.target.closest('.upload-zone');
            if (uploadArea && !uploadArea.contains(e.relatedTarget)) {
                uploadArea.classList.remove('dragover');
            }
        });

        document.addEventListener('drop', function(e) {
            e.preventDefault();
            const uploadArea = e.target.closest('.upload-zone');
            if (uploadArea) {
                uploadArea.classList.remove('dragover');
                const files = e.dataTransfer.files;
                if (files.length > 0) {
                    const fileInput = uploadArea.querySelector('input[type="file"]');
                    if (fileInput) {
                        fileInput.files = files;
                        handleFileUpload(fileInput);
                    }
                }
            }
        });

        // Floating label animation
        document.querySelectorAll('input, textarea').forEach(input => {
            input.addEventListener('focus', function() {
                this.parentElement.classList.add('focused');
            });
            
            input.addEventListener('blur', function() {
                if (!this.value) {
                    this.parentElement.classList.remove('focused');
                }
            });
        });

        // Category expansion functionality
        function toggleCategoryExpansion(category) {
            const extraThemes = document.querySelectorAll(`.${category}-extra`);
            const arrow = document.getElementById(`${category}-arrow`);
            const button = arrow.parentElement;
            
            const isExpanded = !extraThemes[0].classList.contains('hidden');
            
            if (isExpanded) {
                // Collapse
                extraThemes.forEach(theme => {
                    theme.style.opacity = '0';
                    theme.style.transform = 'translateY(-10px)';
                    setTimeout(() => {
                        theme.classList.add('hidden');
                    }, 200);
                });
                arrow.textContent = '→';
                button.innerHTML = `Explore All <span id="${category}-arrow" class="ml-1 transition-transform">→</span>`;
            } else {
                // Expand
                extraThemes.forEach((theme, index) => {
                    theme.classList.remove('hidden');
                    setTimeout(() => {
                        theme.style.opacity = '1';
                        theme.style.transform = 'translateY(0)';
                    }, index * 100);
                });
                arrow.textContent = '↑';
                button.innerHTML = `Show Less <span id="${category}-arrow" class="ml-1 transition-transform">↑</span>`;
            }
        }

        // Progressive form validation
        document.querySelector('form').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Check if a theme is selected
            if (!selectedTheme) {
                // Scroll to theme selection
                document.querySelector('#theme-selection').scrollIntoView({ behavior: 'smooth' });
                
                // Show selection hint
                const themeSection = document.querySelector('.section-card:nth-child(2)');
                themeSection.style.borderColor = '#EF4444';
                setTimeout(() => {
                    themeSection.style.borderColor = 'rgba(255, 255, 255, 0.1)';
                }, 2000);
                return;
            }
            
            // Simulate successful creation with animation
            const button = e.target.querySelector('button[type="submit"]');
            const originalText = button.textContent;
            
            button.innerHTML = `

                Creating Event...

            setTimeout(() => {
                button.innerHTML = `
                    <svg class="mr-2 h-5 w-5 text-white inline" fill="none" stroke="currentColor" viewBox="0 0 24 24">

                button.classList.remove('from-primary-500', 'to-primary-600');
                button.classList.add('from-green-500', 'to-emerald-500');
